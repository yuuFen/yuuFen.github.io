设计模式总结（都在码里）

2020-03-01 3 min read # JS

pub / sub（订阅 / 发布模式，观察者模式）

优点

解耦，降低代码入侵性

原始：
ModuleA
  import B
  B.action()
ModuleB
  action() {
    import C
    C.action()
  }
ModuleC
  action() {
    doSomething()
  }

观察者：
Event
ModuleA
  event.$emit()
ModuleB
  event.$on()
  event.$emit()
ModuleC
  event.$on()

目标 & 实现

参考 vue

// 实现：
class Event {
  constructor() {
    this.callback = {}
  }

  $on(name, fn) {
    ;(this.callback[name] || (this.callback[name] = [])).push(fn)
  }

  $emit(name, ...args) {
    const cbs = this.callback[name]
    if (cbs) {
      cbs.forEach((cb) => {
        cb.call(this, ...args)
      })
    }
  }

  $off(name) {
    this.callback[name] = null
  }
}

// 目标：
let event = new Event()
// 监听
event.$on('event1', (...args) => {
  console.log('event1', ...args)
})
event.$on('event1', (...args) => {
  console.log('event1 again', ...args)
})
// 触发
event.$emit('event1', 1, 2, 3)
// 解绑
console.log('-'.repeat(20))
event.$off('event1')
event.$emit('event1', 1, 2, 3) // 测试

单例模式

保证一个类仅有一个实例，并提供一个访问它的全局访问点。

适用场景

如：

  1. 弹窗，无论点击多少次都只被创建一次。
  2. redux 的 store，全局唯一

简单实例

// 新建弹窗，期望弹窗全局唯一
// 伪代码:
// function createModalLayer() {
//   // 闭包储存单例实例
//   let result
//   return function() {
//     return result || (result = 新建弹窗)
//   }
// }
// 提取：
// 高阶函数：
function getSingle(fn) {
  let result
  return function() {
    return result || (result = fn.apply(this, arguments))
  }
}
function createModalLayer() {
  console.log('新建弹窗')
  const div = document.createElement('div')
  div.innerHTML = new Date()
  div.className = 'modal'
  div.style.display = 'none'
  document.body.appendChild(div)
  div.addEventListener(
    'click',
    function() {
      div.style.display = 'none'
    },
    false,
  )
  return div
}

createModalLayer = getSingle(createModalLayer)

document.getElementById('modal-btn').addEventListener('click', function() {
  // 新建弹窗
  const modalLayer = createModalLayer()
  modalLayer.style.display = 'block'
})

策略模式

分析

KISS 原则 keep it simple and stupid

策略模式指定义一系列算法，把他们一个个封装起来，并使他们可以相互替换。

目的是将算法的使用和实现分离。

一个基于策略模式的程序至少由两部分组成。第一部分是一组策略类（可变），封装了具体的算法，负责具体的计算过程；第二部分是环境类 Context（不变），Context 接收客户的请求，随后将请求委托给某一个策略类。要做到这一点，说明 Context 中要维持对某个策略对象的引用。

实例

原始（面条式代码）：

let form = 获取表单
form.onSubmit = function() {
  if (form.name.value === '') {
    弹出错误消息
  }
  if (form.password.value === '') {
    弹出错误消息
  }
  if (form.password.value.length < 5) {
    弹出错误消息
  }
  正则等等
}

策略模式（具体看 element 的表单验证 https://github.com/ElemeFE/element/blob/dev/packages/form/src/form.vue ），其他例子：

// 1. 策略
// 可以使用配置
const policy = {
  S: function(salary) {
    return salary * 4
  },
  A: function(salary) {
    return salary * 3
  },
  B: function(salary) {
    return salary * 2
  },
  C: function(salary) {
    return salary
  },
}
// 2. 使用
function calculate(level, salary) {
  return policy[level] ? policy[level](salary) : 0
}
